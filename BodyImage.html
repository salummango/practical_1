        import java.util.Scanner;
        public class Salary {
            public static void main(String[] args) {
                int salary;
                int performance;
                Scanner me = new Scanner (System.in);
                System.out.print("ENTER SALARY : " );
                salary=me.nextInt();
                System.out.print("ENTER PERFORMANCE : " );
                performance=me.nextInt();
                if (performance >= 90){
                    int salaryconverter= (salary * 3 / 100) ;
                    int salaryperformance=(salary + salaryconverter);
                    System.out.print("Your salary is : " + salaryperformance);
                }else{
                    System.out.print("Your salary is : " + salary);
                }
            }
        }

The code above deals with with the employee salary that based on their perfomance
and alary will increase up to 3% to only those who have 90% of perfomance
